class Request:
    """
    Parameters for a HTTP request.

    Arguments mimic `asks.request`_, which in turn mimics `requests.request`_.

    Attributes:
       method: HTTP method type `GET`, `OPTIONS`, `HEAD`, `POST`, `PUT`, `PATCH`, or `DELETE`.
       url: URL of the Request
       params: Dictionary, list of tuples or bytes to send in the query string of the Request
       data: Dictionary, list of tuples, bytes, or file-like object to send in the body of the Request
       json: A JSON serializable Python object to send in the body of the Request
       headers: Dictionary of HTTP Headers to send with the Request
       cookies: Dict or CookieJar object to send with the Request
       auth: Auth tuple to enable Basic/Digest/Custom HTTP Auth

    .. _asks.request:
        https://asks.readthedocs.io/en/latest/overview-of-funcs-and-args.html
    .. _requests.request:
        https://2.python-requests.org/en/master/api/#requests.request
    """

    method: str
    url: str
    params = None
    data = None
    json = None
    headers = None
    cookies = None
    auth = None

[docs]    @classmethod
    def gen_requests(
            cls,
            method: Union[str, List[str]],
            url: Union[str, List[str]],
            params=None,
            data=None,
            json=None,
            headers=None,
            cookies=None,
            auth=None,
    ):
        """
        Populate self with list of Requests.

        **Each argument could be a single item or a list of items. When they are a single item, that attribute is
        duplicated for every request. All lists must have the same length.**

        Arguments mimic `asks.request`_, which in turn mimics `requests.request`_.

        Args:
            method: HTTP method type `GET`, `OPTIONS`, `HEAD`, `POST`, `PUT`, `PATCH`, or `DELETE`.
            url: URL of the Request
            params: Dictionary, list of tuples or bytes to send in the query string of the Request
            data: Dictionary, list of tuples, bytes, or file-like object to send in the body of the Request
            json: A JSON serializable Python object to send in the body of the Request
            headers: Dictionary of HTTP Headers to send with the Request
            cookies: Dict or CookieJar object to send with the Request
            auth: Auth tuple to enable Basic/Digest/Custom HTTP Auth

        .. _asks.request:
            https://asks.readthedocs.io/en/latest/overview-of-funcs-and-args.html
        .. _requests.request:
            https://2.python-requests.org/en/master/api/#requests.request
        """

        self = cls()
        locals_ = locals()  # Locals inside dict comprehension is local to the comprehension
        attributes = {attr: locals_[attr] for attr in ATTRIBUTE_NAMES}

        list_attr = []
        single_attr = []
        lengths = []
        for attr, value in attributes.items():
            if isinstance(value, list):
                list_attr.append(attr)
                lengths.append(len(value))
            else:
                single_attr.append(attr)

        if len(list_attr) != 0:
            if [lengths[0]] * len(lengths) != lengths:
                raise ValueError("All lists must be of the same length")
            length = lengths[0]

        else:
            length = 1

        for i in range(length):
            args = {}
            for attr in single_attr:
                args[attr] = attributes[attr]
            for attr in list_attr:
                args[attr] = attributes[attr][i]

            self.append(Request(**args))

        return self

[docs]    def __init__(
            self,
            n_workers=N_WORKERS_DEFAULT,
            n_connections=N_CONNECTIONS_DEFAULT,
            retries=10,
            retry_sleep=3,
            ok_code=(200,),
            check_response_func=None,
            json=False,
    ):
        """
        Dead easy interface for executing many HTTP requests asynchronously.

        Args:
            n_workers: Max number of workers to use. Too many workers will use a lot of memory and increase startup
                time, too few can lead to slower execution
            n_connections: Max number of open connections to have open at once. Each connection spawns a new worker.
            retries: Number of retries to attempt
            retry_sleep: How long in seconds to wait when request fails
            ok_code: A sequence of HTTP status codes to accept as ok. If `any`, all responses will be assumed to be ok
            check_response_func: A function to apply to the response to determine if ok
            json: Parse response body as json and return instead of full Responce object

        Examples:
            >>> requests = ManyRequests.gen_requests('GET', [f'https://example.org/{i}' for i in range(10)])
            >>> responses = EasyRequests(10, 2, retries=2, retry_sleep=0, ok_code=ok_code, json=json)(requests)
        """
        super().__init__(n_workers=n_workers)
        self.n_connections = n_connections
        self.session = None
        self.retries = retries
        self.retry_sleep = retry_sleep
        self.ok_code = ok_code
        self.check_response_func = check_response_func
        self.json = json

        self.requests = None
        self.responses = None



[docs]    def __call__(self, requests: ManyRequests):
        """
        Requests to process asynchronously returns the responses in the same order.
        If no ok response was obtained after retires, a `None` will be included

        Args:
            requests: `ManyRequests` of list of requests to process

        Returns:
            responses: A list of responses in the same order of the requests. Will include `None` in the position
                of a request where no good response was obtained
        """

        self.requests = requests
        self.responses = [None] * len(requests)

        self.session = asks.Session(connections=self.n_connections)

        tasks = [delayed(self._runner)(idx, request) for idx, request in enumerate(requests)]
        super().__call__(tasks)

        return self.responses


    async def _runner(self, idx: int, request: Request):
        """Task which handles completing a HTTP request and errors that arise"""

        last_error = None

        for i in range(self.retries):
            try:
                response = await self.session.request(
                    **{
                        attr: getattr(request, attr)
                        for attr in ATTRIBUTE_NAMES
                    }
                )

                if self.ok_code != "any" and response.status_code not in self.ok_code:
                    raise BadResponse(
                        f"Bad response status code: {response.status_code}. Should be in {self.ok_code}"
                    )

                if self.check_response_func is not None:
                    self.check_response_func(response)

                if self.json:
                    response = response.json()

                logging.debug(f"OK Request {request}")

                self.responses[idx] = response
                return

            except (RemoteProtocolError, BadHttpResponse, JSONDecodeError, BadResponse) as e:
                try:
                    code, text = response.status_code, response.text
                except NameError:
                    code, text = None, None

                logging.info(
                    f"BAD Request {request}: retry {1 + i}. Error {type(e).__name__}. Code: {code}. Body: {text}"
                )

                last_error = e
                await trio.sleep(self.retry_sleep)

        logging.warning(
            f"FAILED Request {request}: Permanently failed. Last error {type(last_error).__name__}"
        )
        self.responses[request] = response, last_error
